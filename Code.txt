document.getElementById('nextDateButton').addEventListener('click', () => {
    if (document.getElementById('datePicker').value) {
        page1.classList.add('hidden');
        page2.classList.add('hidden');
        page3.classList.remove('hidden');
    } else {
        alert('Please select a date!');
    }
});

document.getElementById('nextTypeButton').addEventListener('click', () => {
    if (document.getElementById('dateType').value) {
        page3.classList.add('hidden');
        page4.classList.remove('hidden');
    } else {
        alert('Please select a date type!');
    }
});

document.getElementById('nextLocationButton').addEventListener('click', () => {
    const date = document.getElementById('datePicker').value;
    const type = document.getElementById('dateType').value;
    const location = document.getElementById('location').value;

    const summary = `We’re going on a ${type} on ${date}. I will pick you up at: ${location}.`;
    document.getElementById('summaryText').innerText = summary;

    page4.classList.add('hidden');
    page5.classList.remove('hidden');
});
